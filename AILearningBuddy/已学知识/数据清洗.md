## 步骤3：完整代码展示

```python
# ============================================
# 数据清洗 — 完整可运行代码
# ============================================

import pandas as pd
import numpy as np
from sklearn.impute import SimpleImputer
import re

# ============================================
# 第一部分：认识脏数据
# ============================================

print("=" * 50)
print("第一部分：认识脏数据")
print("=" * 50)

# ① 创建脏数据集
print("\n=== 1. 创建脏数据集 ===")

data = {
    "姓名": ["张三", "李四", None, "王五", "赵六"],
    "年龄": [25, -30, "三十", 150, 28],
    "性别": ["男", "M", "female", "男", None],
    "电话": ["13812345678", "abc", "138-8888-9999", None, "未知"],
    "工资": ["15000", "面议", "8k-15k", np.nan, "20K"],
    "城市": ["北京", "", "上海", None, "深圳"],
    "入职年份": [2020, 2019, 2021, 2020, None]
}

df = pd.DataFrame(data)
print("脏数据 DataFrame:")
print(df)
print(f"\n数据类型:\n{df.dtypes}")

# ============================================
# 第二部分：缺失值处理
# ============================================

print("\n" + "=" * 50)
print("第二部分：缺失值处理")
print("=" * 50)

# ② 查看缺失值
print("\n=== 2. 查看缺失值 ===")

print(f"各列缺失值数量:\n{df.isnull().sum()}")
print(f"\n总缺失率: {df.isnull().sum().sum() / df.size:.2%}")

# ③ 删除缺失值
print("\n=== 3. 删除缺失值 ===")

# 删除整行（有任意缺失值就删）
df_clean = df.dropna()
print(f"删除前: {len(df)}行，删除后: {len(df_clean)}行")

# 删除整列（缺失率超过50%的列）
threshold = len(df) * 0.5
df_clean2 = df.dropna(axis=1, thresh=int(threshold))
print(f"删除缺失率>50%的列后: {df_clean2.shape}")

# 只删除目标列的缺失值
df_no_target_na = df.dropna(subset=["工资"])
print(f"删除工资列缺失值后: {len(df_no_target_na)}行")

# ④ 填充缺失值
print("\n=== 4. 填充缺失值 ===")

# 用常数填充
df_filled = df.copy()
df_filled["年龄"] = df_filled["年龄"].fillna(0)
df_filled["姓名"] = df_filled["姓名"].fillna("未知")
print(f"用0/未知填充:\n{df_filled}")

# 用均值填充（数值列）
df_filled2 = df.copy()
df_filled2["工资"] = pd.to_numeric(df_filled2["工资"], errors='coerce')  # 先转数值
mean_salary = df_filled2["工资"].mean()
df_filled2["工资"] = df_filled2["工资"].fillna(mean_salary)
print(f"\n用均值填充工资: mean={mean_salary:.0f}")

# 用众数填充（类别列）
df_filled3 = df.copy()
most_common_gender = df_filled3["性别"].mode()[0]
df_filled3["性别"] = df_filled3["性别"].fillna(most_common_gender)
print(f"\n用众数填充性别: {most_common_gender}")

# 用前后值填充
df_filled4 = df.copy()
df_filled4["城市"] = df_filled4["城市"].fillna(method='ffill')  # 前值
print(f"\n用前值填充城市:\n{df_filled4['城市']}")

# sklearn 的填充器（更专业）
print("\n用 sklearn SimpleImputer:")
arr = df[["工资", "入职年份"]].copy()
arr["工资"] = pd.to_numeric(arr["工资"], errors='coerce')
arr["入职年份"] = pd.to_numeric(arr["入职年份"], errors='coerce')

imputer = SimpleImputer(strategy='mean')  # mean/median/most_frequent/constant
arr_imputed = imputer.fit_transform(arr)
print(f"均值填充结果:\n{arr_imputed}")

# ============================================
# 第三部分：重复值处理
# ============================================

print("\n" + "=" * 50)
print("第三部分：重复值处理")
print("=" * 50)

# ⑤ 检测和删除重复
print("\n=== 5. 重复值处理 ===")

df_dup = pd.DataFrame({
    "姓名": ["张三", "李四", "张三", "王五", "李四", "赵六"],
    "年龄": [25, 30, 25, 35, 30, 28],
    "城市": ["北京", "上海", "北京", "深圳", "上海", "广州"]
})

print(f"有重复的数据:\n{df_dup}")

# 检测重复
print(f"\n重复行:\n{df_dup.duplicated()}")
print(f"姓名重复:\n{df_dup.duplicated(subset=['姓名'])}")

# 删除重复（默认保留第一个）
df_no_dup = df_dup.drop_duplicates()
print(f"\n删除重复后:\n{df_no_dup}")

# 保留最后一个
df_no_dup_last = df_dup.drop_duplicates(keep='last')
print(f"\n保留最后一个:\n{df_no_dup_last}")

# ============================================
# 第四部分：格式统一化
# ============================================

print("\n" + "=" * 50)
print("第四部分：格式统一化")
print("=" * 50)

# ⑥ 字符串标准化
print("\n=== 6. 字符串标准化 ===")

# 统一性别格式
df["性别_统一"] = df["性别"].str.lower()  # 转小写
df["性别_统一"] = df["性别_统一"].replace({
    "male": "男", "m": "男", "female": "女", "f": "女", "男同胞": "男"
})
df["性别_统一"] = df["性别_统一"].fillna("未知")
print(f"统一性别:\n{df[['性别', '性别_统一']]}")

# ⑦ 数值提取
print("\n=== 7. 数值提取 ===")

# 从字符串中提取数字
def extract_number(val):
    if pd.isna(val):
        return np.nan
    val = str(val)
    # 匹配数字（包括小数和负数）
    match = re.search(r'-?\d+\.?\d*', val.replace(',', ''))
    if match:
        return float(match.group())
    return np.nan

df["工资_数值"] = df["工资"].apply(extract_number)
print(f"提取工资数值:\n{df[['工资', '工资_数值']]}")

# ⑧ 数据类型转换
print("\n=== 8. 数据类型转换 ===")

# 转数值类型
df["年龄_数值"] = pd.to_numeric(df["年龄"], errors='coerce')
print(f"年龄转数值:\n{df[['年龄', '年龄_数值']]}")

# 转日期类型
df["入职日期"] = pd.to_datetime(df["入职年份"], format="%Y", errors='coerce')
print(f"\n入职年份转日期:\n{df[['入职年份', '入职日期']]}")

# ============================================
# 第五部分：异常值处理
# ============================================

print("\n" + "=" * 50)
print("第五部分：异常值处理")
print("=" * 50)

# ⑨ 识别异常值
print("\n=== 9. 识别异常值 ===")

ages = df[["年龄"]].copy()
ages["年龄_数值"] = pd.to_numeric(ages["年龄"], errors='coerce')

# 方法1：IQR法
Q1 = ages["年龄_数值"].quantile(0.25)
Q3 = ages["年龄_数值"].quantile(0.75)
IQR = Q3 - Q1
lower = Q1 - 1.5 * IQR
upper = Q3 + 1.5 * IQR

print(f"正常年龄范围: [{lower}, {upper}]")
print(f"异常值:\n{ages[(ages['年龄_数值'] < lower) | (ages['年龄_数值'] > upper)]}")

# 方法2：Z-score法
mean_age = ages["年龄_数值"].mean()
std_age = ages["年龄_数值"].std()
ages["z_score"] = (ages["年龄_数值"] - mean_age) / std_age
print(f"\n|Z-score| > 3 为异常:\n{ages[abs(ages['z_score']) > 3]}")

# ⑩ 处理异常值
print("\n=== 10. 处理异常值 ===")

# 方案1：删除
df_clean = df.copy()
df_clean["年龄_数值"] = pd.to_numeric(df_clean["年龄"], errors='coerce')
df_clean = df_clean[(df_clean["年龄_数值"] >= 0) & (df_clean["年龄_数值"] <= 120)]
print(f"删除年龄异常值后:\n{df_clean[['姓名', '年龄', '年龄_数值']]}")

# 方案2：用边界值替代
df_fixed = df.copy()
df_fixed["年龄_数值"] = pd.to_numeric(df_fixed["年龄"], errors='coerce')
df_fixed.loc[df_fixed["年龄_数值"] < 0, "年龄_数值"] = abs(df_fixed.loc[df_fixed["年龄_数值"] < 0, "年龄_数值"])  # 负数取绝对值
df_fixed.loc[df_fixed["年龄_数值"] > 120, "年龄_数值"] = 120  # 超大值设为120
print(f"\n修正年龄异常值后:\n{df_fixed[['姓名', '年龄', '年龄_数值']]}")

# ============================================
# 第六部分：数据替换
# ============================================

print("\n" + "=" * 50)
print("第六部分：数据替换")
print("=" * 50)

# ⑪ 替换特定值
print("\n=== 11. 替换特定值 ===")

df_replace = df.copy()

# 替换空字符串和"暂无"为NaN
df_replace["城市"] = df_replace["城市"].replace(["", "暂无", "未知"], np.nan)
print(f"替换空字符串后:\n{df_replace[['姓名', '城市']]}")

# 替换电话号码中的非数字
df_replace["电话_纯数字"] = df_replace["电话"].apply(
    lambda x: re.sub(r'\D', '', str(x)) if pd.notna(x) else np.nan
)
print(f"\n电话号码提取数字:\n{df_replace[['电话', '电话_纯数字']]}")

# ============================================
# 第七部分：数据验证
# ============================================

print("\n" + "=" * 50)
print("第七部分：数据验证")
print("=" * 50)

# ⑫ 数据清洗后验证
print("\n=== 12. 数据清洗流程总结 ===")

def clean_data(df):
    """完整的数据清洗流程"""
    df = df.copy()
    
    # 1. 统一性别
    gender_map = {"male": "男", "m": "男", "female": "女", "f": "女", "男同胞": "男"}
    df["性别"] = df["性别"].str.lower().replace(gender_map)
    
    # 2. 提取工资数值
    def extract_salary(x):
        if pd.isna(x):
            return np.nan
        match = re.search(r'\d+', str(x).replace(',', ''))
        return float(match.group()) if match else np.nan
    df["工资"] = df["工资"].apply(extract_salary)
    
    # 3. 提取年龄数值
    df["年龄"] = pd.to_numeric(df["年龄"], errors='coerce')
    # 年龄异常值处理：负数取绝对值，超过150的设为NaN
    df.loc[df["年龄"] < 0, "年龄"] = abs(df.loc[df["年龄"] < 0, "年龄"])
    df.loc[df["年龄"] > 150, "年龄"] = np.nan
    
    # 4. 处理城市空值
    df["城市"] = df["城市"].replace(["", "暂无", "未知"], np.nan)
    
    # 5. 删除重复行
    df = df.drop_duplicates()
    
    # 6. 填充剩余缺失值
    df["城市"] = df["城市"].fillna(df["城市"].mode()[0] if not df["城市"].mode().empty else "未知")
    df["工资"] = df["工资"].fillna(df["工资"].mean())
    df["年龄"] = df["年龄"].fillna(df["年龄"].median())
    
    return df

df_final = clean_data(df)
print("清洗后的数据:")
print(df_final)
print(f"\n清洗后数据类型:\n{df_final.dtypes}")

# ============================================
# 第八部分：数据清洗速查
# ============================================

print("\n" + "=" * 50)
print("第八部分：数据清洗速查")
print("=" * 50)

print("""
=== 缺失值处理 ===
  df.isnull().sum()                    - 查看缺失值
  df.dropna()                          - 删除缺失行
  df.fillna(value)                     - 用值填充
  df.fillna(method='ffill')            - 用前/后值填充
  SimpleImputer(strategy='mean')        - sklearn均值填充

=== 重复值处理 ===
  df.duplicated()                       - 检测重复
  df.drop_duplicates()                  - 删除重复
  df.drop_duplicates(subset=['col'])   - 按列删除重复

=== 格式统一 ===
  str.lower() / str.upper()             - 大小写转换
  str.strip()                           - 去除空格
  replace({'a': 'b', 'c': 'd'})        - 批量替换
  pd.to_numeric(errors='coerce')         - 转数值（失败变NaN）
  pd.to_datetime()                      - 转日期

=== 异常值处理 ===
  quantile(0.25/0.75)                   - 计算分位数
  IQR = Q3 - Q1                         - IQR法
  z-score = (x - mean) / std           - Z-score法
  df[(x > lower) & (x < upper)]       - 保留正常范围

=== 数据验证 ===
  df.describe()                         - 基本统计
  df.info()                             - 数据类型/缺失
  df['col'].value_counts()             - 值分布
""")
```

## 步骤4：逐行/逐模块代码详解

### 模块①：缺失值处理策略

| 策略 | 适用场景 | 方法 |
|------|---------|------|
| 删除 | 缺失比例小 (<5%) | `dropna()` |
| 均值填充 | 数值列，无极端值 | `fillna(mean)` |
| 中位数填充 | 数值列，有极端值 | `fillna(median)` |
| 众数填充 | 类别列 | `fillna(mode[0])` |
| 前/后值填充 | 时间序列数据 | `ffill` / `bfill` |
| 预测填充 | 缺失较多 | 用模型预测 |

### 模块②：异常值检测方法

**IQR法（四分位距）**：
```
正常范围 = [Q1 - 1.5×IQR, Q3 + 1.5×IQR]
```

**Z-score法**：
```
|Z| > 3 → 异常
```

### 模块③：正则表达式提取

```python
# \d+  - 匹配一个或多个数字
# \D   - 匹配非数字
# -?\d+\.?\d*  - 匹配整数或小数，可带负号

re.search(r'\d+', "15000")  # → '15000'
re.sub(r'\D', '', "138-8888")  # → '1388888'
```

### 模块④：清洗流程标准顺序

```
1. 统一格式（大小写、空格）
2. 统一编码
3. 处理缺失值
4. 处理重复值
5. 处理异常值
6. 数据类型转换
7. 特征工程
8. 验证
```

## 步骤7：避坑提示与知识点关联

### ⚠️ 避坑提示

| 坑 | 错误 | 正确 |
|---|------|------|
| 直接删缺失 | 缺失多就删 | 先分析缺失原因和比例 |
| 用均值填所有 | 对有异常值的列用均值 | 先处理异常值或用中位数 |
| 字符串不清理 | " 男 " 和 "男" 不同 | `str.strip()` 去空格 |
| 异常值直接删 | 影响数据量 | 优先考虑修正，或标记后单独分析 |
